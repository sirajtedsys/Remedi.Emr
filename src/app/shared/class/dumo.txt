import { Component, EventEmitter, Input, OnChanges, OnInit, Output } from '@angular/core';
import { CommonModule, DatePipe } from '@angular/common';
import { FormsModule } from '@angular/forms';
import { IonButton, IonIcon } from '@ionic/angular/standalone';
import { ProcedureNote } from 'src/app/shared/class/remedi.cc/procedure-note/procedure-note';
import { LookUp } from 'src/app/shared/interfaces/lookup/lookup';
import { ProcedureNoteService } from 'src/app/shared/services/remedi.cc/procedure-note.service';
import { NotificationService } from 'src/app/shared/services/notification/notification.service';
import { firstValueFrom } from 'rxjs';
import { TherapistPrcNote } from 'src/app/shared/interfaces/remedi.cc/procedure-notes/therapist-prc-notes';

@Component({
  selector: 'app-procedure-note',
  templateUrl: './procedure-note.page.html',
  styleUrls: ['./procedure-note.page.scss'],
  standalone: true,
  imports: [IonIcon, IonButton, CommonModule, FormsModule]
})
export class ProcedureNotePage implements OnInit, OnChanges {

  @Input() ProcedureList: LookUp[] = []
  @Input() MachineList: LookUp[] = []
  @Input() pn = new ProcedureNote()
  @Output() close = new EventEmitter<any>();




  constructor(
    private prcService: ProcedureNoteService,
    private notificationService: NotificationService,
    private datePipe: DatePipe
  ) { }

  ngOnInit() {
  }


  ngOnChanges() {
    console.log(this.pn)
    if (this.pn.NoteId == 0) {
      // this.pn.prC_ID=this.ProcedureList[0].id
      this.GetTherapistNote(this.pn.PrcId, this.pn.BillHdrId)

    }
  }


  async GetTherapistNote(prcId: any, BillhdrId: any) {
    (await this.prcService.GetTherapistNote(prcId,BillhdrId)).subscribe(res => {
      console.log(res)
      if(res.length==0)
      {
        return
      }
      let data:TherapistPrcNote=res[0]
      this.pn.Comments=data.comments??''
      this.pn.Note=data.note
      this.pn.StartDate= this.datePipe.transform(data.startDate, 'yyyy-MM-dd') ?? data.startDate
      this.pn.EndDate=data.endDate
      this.pn.PrcBkMachineId=(data.prcBkMachineId??0).toString();
    })
  }



async saveProcedureNote() {

  try {

    if(this.pn.StartDate==null)
    {
      this.notificationService.showNotification('Please select start date', 'error');
      return
    }

    if(this.pn.Note==null || this.pn.Note=='')
    {
      this.notificationService.showNotification('Please Enter Note', 'error');
      return
    }

    let data: { message: string; status: number };

    if (!this.pn.NoteId || this.pn.NoteId === 0) {

      data = await firstValueFrom(
        await this.prcService.insertProcedureNote(this.pn)
      );

    } else {

      data = await firstValueFrom(
        await this.prcService.UpdateProcedureNote(this.pn)
      );

    }

    console.log(this.pn);

    if (data.status === 200) {
      this.notificationService.showNotification(data.message, 'success');
      this.closeModal();
    } else {
      this.notificationService.showNotification(data.message, 'error');
    }

  } catch (error) {
    console.error(error);
    this.notificationService.showNotification('Something went wrong', 'error');
  }
}


clearProcedureNote() {
  this.pn.Note = '';
  this.pn.Comments = '';
  this.pn.StartDate = null;
  this.pn.EndDate = null;
  this.pn.PrcBkMachineId = '';
}


closeModal() {
  this.close.emit({ success: false });
}



}
